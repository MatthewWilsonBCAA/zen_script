assign x 0
assign y 0
assign z 0
assign MAX_X 5
assign MAX_Y 5
assign MAX_Z 5
if $x <= $MAX_X
  com
  if $y <= $MAX_Y
    assign y math $y + 1
    if $z <= $MAX_Z
        display X: $x , Y: $y , Z: $z
        assign z math $z + 1
        repeat
    end
    assign z 0
    repeat
  end
  assign x math $x + 1
  assign y 0
  repeat
end
